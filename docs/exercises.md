### Date in local time

```js
const date = new Date(2020, 9, 22);
const options = { month: "2-digit", day: "2-digit" };
console.log(date.toLocaleDateString("en", options)); // 10/22
console.log(date.toLocaleDateString("pl", options)); // 22.10
```
